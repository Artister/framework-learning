object WordCount {

	def main (args: Array[String]) {
		val conf = new SparkConf().setAppName("wordcount").
		var sc = new SparkContext(conf)
		
		val input = sc.textFile("/home/maixia/soft/heloSpark.txt")
		
		val lines = input.flatMap(line =>line.split(" "))
		val count = lines.map(word=>(word, 1)).ruduceByKey(case (x, y)=>x+y)
		
		val output = count.saveAsTextFile("/home/maixia/soft/heloSparkRes")
		
	}


}
